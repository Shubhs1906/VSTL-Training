package com.popupFactory;

import org.openqa.selenium.By;

import com.generic.Pojo;


public class SelectFolderViewPopupPageByShubhamM {
	private Pojo objPojo;
	

	public SelectFolderViewPopupPageByShubhamM(Pojo pojo) {
		this.objPojo = pojo;
	}

	By loc_hdrSelectFolderView = By.xpath("//div[@role='dialog']/following::span[contains(text(),'Select Folder View')]");
	By loc_btnSOTView =By.xpath("//button[@id='btnSOTView'][text()='SOT View']");
	

	
	public void verifySelectFolderViewPopupPageIsDisplayed() {	
		objPojo.getObjUtilities().logReporter("Verify user can see 'Select Folder View' popup page is displayed", 
				"User should be able to see the 'Select Folder View' popup page header", 
				"User can is able to see the 'Select Folder View' popup page header", 
				objPojo.getObjWrapperFunctions().checkElementDisplayed(loc_hdrSelectFolderView));	
	}
	public void ClickOnSOTViewButton() {
		objPojo.getObjUtilities().logReporter("Verify user can click on 'SOT View' button",
				"User should be able to click on 'SOT View' button",
				"User is able to click on 'SOT View' button",
				objPojo.getObjWrapperFunctions().click(loc_btnSOTView));
	}
	


}
