package com.pageFactory;

import org.openqa.selenium.By;

import com.generic.Pojo;

public class DashboardPageByShubhamM {

	// Global Variables
	private Pojo objPojo;
	

	// Constructor of class
	public DashboardPageByShubhamM(Pojo pojo) {
		this.objPojo = pojo;
	}

	

	
	
	public void clickOnMenuIconFromLeftSideBarMenu(String strMenuIcon) {
		By locator = By.xpath("//li//a//span[contains(text(),'" + strMenuIcon + "')]/preceding-sibling::i");
		objPojo.getObjWrapperFunctions().checkElementDisplayed(locator);
		objPojo.getObjUtilities().logReporter("Verify user can click on menu icon",
				"User should be able to click on menu icon '"+strMenuIcon+"'",
				"User is able to click on menu icon '"+strMenuIcon+"'", 
				objPojo.getObjWrapperFunctions().click(locator));
	}
	
	
	public void verifyPageHeaderIsDisplayed(String strPageHeader) {
		By locator = By.xpath("//body//div//h1[contains(text(),'" + strPageHeader + "')]");
		objPojo.getObjWrapperFunctions().checkElementDisplayed(locator);
		objPojo.getObjUtilities().logReporter("Verify '" + strPageHeader + "' page header is displayed.",
				objPojo.getObjWrapperFunctions().checkElementDisplayed(locator));
	}

}
