
        private static XSSFWorkbook workbook;
	private static XSSFSheet excelsheet;
	private static XSSFRow row;
	private static XSSFCell cell;





public void verifyColourOfCell(String excelFileName, String strSheetName) {

		// Excel file load
		// String strTestData="";
		int lastRowNum = 0;
		// XSSFRow row;
		String excelFilePath = System.getProperty("user.dir") + "/src/test/resources/downloads/" + excelFileName
				+ ".xlsx";
		try {
			FileInputStream excelFile = new FileInputStream(excelFilePath);
			workbook = new XSSFWorkbook(excelFile);
			excelsheet = workbook.getSheet(strSheetName);

			lastRowNum = excelsheet.getLastRowNum();
			System.out.println(lastRowNum);

		} catch (Exception e) {

			System.out.println("While loading excel getting error >>" + e.getMessage());
			e.printStackTrace();
		}

		for (int intRow = 1; intRow < lastRowNum; intRow++) {
			row = (XSSFRow) excelsheet.getRow(intRow);
			for (int intCol = 0; intCol < row.getLastCellNum(); intCol++) {

				try {
					if (!excelsheet.getRow(intRow).getCell(intCol).equals(""))
						cell = excelsheet.getRow(intRow).getCell(intCol);

					XSSFCellStyle cs = cell.getCellStyle();
					String strColourCode = cs.getFillForegroundColorColor().getARGBHex();
					System.out.println("Cell Colour Code : " + strColourCode);

				} catch (Exception exception) {

					System.out.println("Getting error while getting cell colour :" + exception.getMessage());
					exception.printStackTrace();

				}
			}
		}
	}
